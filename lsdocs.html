public func default(type) {
    return if (type === bool  ) false
      else if (type === byte  ) 0b
      else if (type === short ) 0s
      else if (type === int   ) 0
      else if (type === long  ) 0L
      else if (type === float ) 0f
      else if (type === double) 0d
      else if (type === str   ) ""
      else if (type === array ) []
      else if (type === void  ) none
      else fail;
}
